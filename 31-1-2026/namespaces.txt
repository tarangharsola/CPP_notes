/*
 WHAT ARE NAMESPACES?
 * 
 * Think of namespaces like Java packages, but more flexible:
 * 
 * Java:  package com.example;
 * C++:   namespace example { ... }
 * 
 * PURPOSE: Prevent name conflicts
 * - Different libraries might both have a "String" class
 * - Namespaces keep them separate
 */

// Example: Creating your own namespace
namespace MyLibrary {
    int calculate(int x) {
        return x * 2;
    }
}

namespace OtherLibrary {
    int calculate(int x) {
        return x * 3;
    }
}
#include<iostream>
void demonstrateNamespaces() {
    // Without 'using namespace', you need to specify which one:
    std::cout << MyLibrary::calculate(5) << std::endl;     // Prints 10
    std::cout << OtherLibrary::calculate(5) << std::endl;  // Prints 15
}

/*
 * THE :: OPERATOR (Scope Resolution Operator)
 * 
 * USES OF ::
 * 1. Access namespace members: std::cout
 * 2. Access class static members: MyClass::staticMethod()
 * 3. Define methods outside class: void MyClass::method() { }
 * 4. Access global scope: ::globalVariable
 * 
 * Examples:
 */

int globalVar = 100;

namespace Demo {
    int globalVar = 200;
    
    void showScope() {
        int globalVar = 300;  // Local variable
        
        std::cout << globalVar << std::endl;        // 300 (local)
        std::cout << Demo::globalVar << std::endl;  // 200 (namespace)
        std::cout << ::globalVar << std::endl;      // 100 (global)
    }
}



// ============================================
// #include vs import - CRITICAL DIFFERENCE!
// ============================================

/*
 * JAVA import:
 * - Tells compiler where to find classes
 * - Doesn't copy any code
 * - Just makes names available
 * - Example: import java.util.ArrayList;
 * 
 * C++ #include:
 * - LITERALLY COPIES the entire file
 * - Preprocessor directive (runs before compilation)
 * - Text substitution
 * - Example: #include <iostream>
 * 
 * HOW #include WORKS:
 * 
 * When you write:
 *   #include <iostream>
 * 
 * The preprocessor:
 * 1. Finds the iostream file
 * 2. COPIES ALL of its contents
 * 3. PASTES it where the #include line was
 * 4. Then compilation begins
 * 
 * DEMONSTRATION:
 */

// If you had a file "myheader.h" with:
// int add(int a, int b) { return a + b; }

// And you write:
// #include "myheader.h"

// The preprocessor literally replaces that line with:
// int add(int a, int b) { return a + b; }

/*
 * INCLUDE TYPES:
 * 
 * #include <filename>  // System/library headers (angle brackets)
 * - Searches standard system directories
 * - For standard library and installed libraries
 * - Example: <iostream>, <string>, <vector>
 * 
 * #include "filename"  // User headers (quotes)
 * - Searches current directory first
 * - For your own header files
 * - Example: "myclass.h", "utilities.h"
 * 
 * Caution to developers => prevent multiple inclusion
 * 
 * 
 * How to avoid ?? 
 * In myheader.h:
 * #ifndef MYHEADER_H
 * #define MYHEADER_H
 * 
 * // header contents here
 * 
 * #endif
 * 
 * This prevents the same code from being copied twice!
 */
